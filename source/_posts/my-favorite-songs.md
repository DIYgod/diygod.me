<script>
$(function () {
    $.ajax({
        url: 'https://api.i-meto.com/meting/api?server=netease&type=playlist&id=35798529',
        success: function (list) {
            var ap = new APlayer({
                element: document.getElementById('aplayer1'),
                showlrc: 3,
                theme: '#ad7a86',
                listmaxheight: '280px',
                mode: 'random',
                music: list
            });
            window.aplayers || (window.aplayers = []);
            window.aplayers.push(ap);
        }
    })
})
</script>
